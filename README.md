   <script>
    function loadAdminData() {
        // 1. Fetch Deposits from Server
        fetch('http://localhost:3000/api/admin/deposits')
            .then(res => res.json())
            .then(data => {
                const tbody = document.getElementById('depositTableBody');
                tbody.innerHTML = ''; 
                
                if(data.length === 0) {
                    tbody.innerHTML = '<tr><td colspan="4" style="text-align:center; color:#777;">No deposits logged yet.</td></tr>';
                }

                data.forEach(item => {
                    // Status ke mutabiq alag color class set karne ke liye
                    let statusClass = 'pending';
                    if(item.status === 'Approved') statusClass = 'approved';
                    if(item.status === 'Rejected') statusClass = 'rejected'; // Iske liye CSS niche di hai

                    // Buttons sirf tabhi dikhein jab status 'Pending' ho
                    let actionButtons = '';
                    if(item.status === 'Pending') {
                        actionButtons = `
                            <button onclick="updateStatus(${item.id}, 'Approved')" style="background:#25D366; color:#000; padding:5px 10px; border:none; border-radius:5px; font-weight:bold; cursor:pointer; margin-right:5px;">Approve</button>
                            <button onclick="updateStatus(${item.id}, 'Rejected')" style="background:#ff3c3c; color:#fff; padding:5px 10px; border:none; border-radius:5px; font-weight:bold; cursor:pointer;">Reject</button>
                        `;
                    } else {
                        actionButtons = `<span style="color:#555; font-size:14px;">No action needed</span>`;
                    }

                    tbody.innerHTML += `
                        <tr>
                            <td>#${item.id}</td>
                            <td style="font-weight:bold; color:#25D366;">Rs. ${item.amount}</td>
                            <td style="color:#ddd; font-family:monospace;">${item.trxid}</td>
                            <td><span class="status ${statusClass}">${item.status}</span></td>
                            <td>${actionButtons}</td>
                        </tr>
                    `;
                });
            })
            .catch(err => console.error("Error loading deposits:", err));

        // 2. Fetch Subscribers from Server
        fetch('http://localhost:3000/api/admin/subscribers')
            .then(res => res.json())
            .then(data => {
                const list = document.getElementById('subscriberList');
                list.innerHTML = ''; 

                if(data.length === 0) {
                    list.innerHTML = '<li style="color:#777; border-left-color:#444;">No email subscribers yet.</li>';
                }

                data.forEach(sub => {
                    list.innerHTML += `<li><i class="fas fa-user"></i> ${sub.email}</li>`;
                });
            })
            .catch(err => console.error("Error loading subscribers:", err));
    }

    // NEW: STATUS UPDATE CALL BACK FUNCTION
    function updateStatus(id, newStatus) {
        fetch('http://localhost:3000/api/admin/update-status', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({ id: id, status: newStatus })
        })
        .then(res => res.json())
        .then(data => {
            if(data.success) {
                alert(`Deposit #${id} has been ${newStatus}!`);
                loadAdminData(); // Table data refresh karne ke liye dobara call karein
            }
        })
        .catch(err => console.error("Error updating status:", err));
    }

    window.onload = loadAdminData;
</script>
